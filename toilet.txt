@app.route("/additional_cash", methods=["GET", "POST"])
@login_required
def additional_cash():
    """Adding Additional Cash"""
    if request.method == "GET":
        return render_template("additional_cash.html")
    else:
        add_cash = int((request.form.get("add_cash")))

        if not add_cash:
            return apology("Insufficient Money", 400)

        user_id = session["user_id"]
        user_cash = db.execute("SELECT cash FROM users WHERE id = ?", user_id)
        cash = user_cash[0]["cash"]

        cash_now = cash + add_cash

        db.execute("UPDATE users SET cash = ? WHERE id = ?", cash_now, user_id)

        return redirect("/")
